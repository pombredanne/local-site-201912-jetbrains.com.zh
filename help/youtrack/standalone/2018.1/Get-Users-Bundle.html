Get Users Bundle

GET /admin/customfield/userBundle/{bundleName}

Get custom field user bundle by its name

GET /rest/admin/customfield/userBundle/{bundleName}

Parameters:

Name

Type

Description

bundleName

string

Name of a user bundle to get from YouTrack.

Sample

Request

GET https://example.com/youtrack/rest/admin/customfield/userBundle/My%20Users Cookie: $Version=0; JSESSIONID=19eie47i6xtd6l37im4l7uxwc; $Path=/; jetbrains.charisma.main.security.PRINCIPAL=NDgxMzQ5NGQxMzdlMTYzMWJiYTMwMWQ1YWNhYjZlN2JiN2FhNzRjZTExODVkNDU2NTY1ZWY1MWQ3Mzc2NzdiMjpyb290; $Path=/

Response

HTTP/1.1 200 OK Content-Type: application/xml;charset=UTF-8 Content-Length: 528 <?xml version="1.0" encoding="UTF-8" standalone="yes"?> <userBundle name="My Users"> <user login="ajordens" url="https://example.com/youtrack/rest/admin/user/ajordens"/> <user login="beto" url="https://example.com/youtrack/rest/admin/user/beto"/> <userGroup name="idea-developers" url="https://example.com/youtrack/rest/admin/group/idea-developers"/> <userGroup name="jira-users" url="https://example.com/youtrack/rest/admin/group/jira-users"/> </userBundle>
