GET Custom Field Types

GET /admin/customfield/type

Get all available custom field types.

GET /rest/admin/customfield/type

Sample

Request:

GET /rest/admin/customfield/type

Response:

<customFieldTypeRefs> <customFieldType name="date" url="http://youtrack.jetbrains.net/rest/admin/customfield/type/date"/> <customFieldType name="integer" url="http://youtrack.jetbrains.net/rest/admin/customfield/type/integer"/> <customFieldType name="string" url="http://youtrack.jetbrains.net/rest/admin/customfield/type/string"/> <customFieldType name="enum[*]" url="http://youtrack.jetbrains.net/rest/admin/customfield/type/enum%5B*%5D"/> <customFieldType name="enum[1]" url="http://youtrack.jetbrains.net/rest/admin/customfield/type/enum%5B1%5D"/> </customFieldTypeRefs>
